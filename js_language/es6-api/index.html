getters and setters

            var obj = {
              get foo() {
                console.log({ name: 'foo', object: obj, type: 'get' });
                return obj._foo;
              },
              set bar(val) {
                console.log({ name: 'bar', object: obj, type: 'set', oldValue: obj._bar });
                return obj._bar = val;
              }
            };
            
            obj.bar = 2;
            // {name: 'baz', object: , type: 'set', oldValue: undefined}
            
            obj.foo;
            // {name: 'foo', object: , type: 'get'}
